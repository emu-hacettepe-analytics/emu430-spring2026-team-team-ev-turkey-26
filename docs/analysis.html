normalize_name <- function(x) {
  x |>
    stri_trans_general("Latin-ASCII") |>
    str_to_lower() |>
    str_replace_all("[^a-z]", "") |>
    str_trim()
}

# ---- 4. Build analysis-ready dataset ----------------------------------------
analysis_ready <- final_df |>
  mutate(
    ev_per_station  = ev_sayisi / istasyon_sayisi,
    ev_market_share = (ev_sayisi / otomobil_sayisi) * 100,
    date            = as.Date(paste0(str_replace(tarih, "_", "-"), "-01")),
    province_key    = normalize_name(sehir)
  ) |>
  arrange(sehir, date)

# ---- 4.5 Hotfix: source-data anomaly in November 2025 -----------------------
analysis_ready <- analysis_ready |>
  group_by(sehir) |>
  arrange(date) |>
  mutate(
    ev_sayisi = if_else(
      ev_sayisi < lag(ev_sayisi) * 0.5 & ev_sayisi < lead(ev_sayisi) * 0.5,
      round((lag(ev_sayisi) + lead(ev_sayisi)) / 2),
      ev_sayisi
    )
  ) |>
  ungroup()

# ---- 5. Correlation: EV adoption vs. station investment ---------------------
global_cor <- cor(analysis_ready$ev_sayisi,
                  analysis_ready$istasyon_sayisi,
                  use = "complete.obs")

3. 🎯 2027 Projections & Investment Gaps

We forecasted the 2027 EV count using a Log-Linear Model and identified provincial investment gaps based on an optimal benchmark of 50 EVs per station.

Province Tier Action Required by 2027
🔴 Critical Risk Immediate capital injection; infrastructure is severely lagging.
🟠 High Risk Infrastructure expansion must double to meet forecasted demand.
🟢 Adequate Current growth pace is sufficient.
Click to View 2027 Forecast & Gap Analysis Code 
# ---- Forecast EVs to 2027 ---------------------------------------------------
target_date_num <- as.numeric(as.Date("2027-01-01"))

forecast_2027 <- analysis_ready |>
  group_by(sehir, province_key) |>
  group_modify(~{
    df <- .x
    if (nrow(df) < 3 || length(unique(df$ev_sayisi)) < 2) {
      return(tibble(ev_2027_linear = NA_real_,
                    ev_2027_loglin = NA_real_,
                    cagr_pct       = NA_real_))
    }
    lin <- lm(ev_sayisi ~ as.numeric(date), data = df)
    pred_lin <- as.numeric(predict(lin, newdata = data.frame(date = target_date_num)))

    if (any(df$ev_sayisi <= 0)) {
      pred_log <- NA_real_; cagr <- NA_real_
    } else {
      llm <- lm(log(ev_sayisi) ~ as.numeric(date), data = df)
      slope <- coef(llm)[2]
      pred_log <- exp(as.numeric(predict(llm, newdata = data.frame(date = target_date_num))))
      cagr <- (exp(slope * 365) - 1) * 100
    }
    tibble(ev_2027_linear = round(pmax(pred_lin, 0)),
           ev_2027_loglin = round(pmax(pred_log, 0)),
           cagr_pct       = round(cagr, 1))
  }) |>
  ungroup() |>
  mutate(ev_2027 = coalesce(ev_2027_loglin, ev_2027_linear))

# ---- Priority scoring -------------------------------------------------------
TARGET_EV_PER_STATION <- 50

priority_table <- forecast_2027 |>
  inner_join(latest_snapshot, by = c("sehir", "province_key")) |>
  mutate(
    stations_needed_2027  = ceiling(ev_2027 / TARGET_EV_PER_STATION),
    investment_gap        = pmax(stations_needed_2027 - current_stations, 0),
    bottleneck_ratio_2027 = ev_2027 / pmax(current_stations, 1),
    priority_tier = case_when(
      ev_2027 > 15000 & bottleneck_ratio_2027 > 100 ~ "Critical",
      ev_2027 > 5000  & bottleneck_ratio_2027 > 75  ~ "High",
      bottleneck_ratio_2027 > 50                    ~ "Medium",
      TRUE                                          ~ "Adequate"
    ),
    priority_tier = factor(priority_tier,
                           levels = c("Critical", "High", "Medium", "Adequate"))
  ) |>
  arrange(desc(investment_gap))

# ---- PLOT 2: Supply-vs-demand gap (dumbbell chart) --------------------------
gap_plot_data <- priority_table |>
  slice_max(investment_gap, n = 15) |>
  mutate(sehir = fct_reorder(sehir, investment_gap))

p2 <- ggplot(gap_plot_data) +
  geom_segment(aes(x = current_stations, xend = stations_needed_2027,
                   y = sehir, yend = sehir),
               color = "grey70", linewidth = 1.2) +
  geom_point(aes(x = current_stations,     y = sehir,
                 color = "Current stations"), size = 4) +
  geom_point(aes(x = stations_needed_2027, y = sehir,
                 color = "Stations needed by 2027"), size = 4) +
  scale_color_manual(
    values = c("Current stations"        = "#1f77b4",
               "Stations needed by 2027" = "#d62728"),
    name = NULL) +
  scale_x_continuous(labels = label_comma()) +
  labs(
    title    = "Charging Infrastructure Gap by 2027",
    subtitle = paste0("Top-15 provinces by investment gap  ·  benchmark = ",
                      TARGET_EV_PER_STATION, " EVs per station"),
    x        = "Number of charging stations",
    y        = NULL,
    caption  = "Forecast: log-linear model on monthly EPDK data"
  ) +
  theme_ev() +
  theme(legend.position = "top")

print(p2)

AVG_STATION_COST_USD <- 50000

map_plot_data_interactive <- map_plot_data |>
  mutate(
    investment_cost_usd = investment_gap * AVG_STATION_COST_USD,
    cost_label = scales::dollar(investment_cost_usd, prefix = "$"),

    weight      = 1,
    opacity     = 1,
    color       = "white",
    dashArray   = "3",
    fillOpacity = 0.8,
    highlightOptions = highlightOptions(
      weight = 3, color = "#666", dashArray = "", fillOpacity = 0.9, bringToFront = TRUE
    ),
    label = ~paste0(
      province_raw,
      " | 2027 EV Forecast: ", ev_label,
      " | Missing Stations: ", investment_gap,
      " | Required Investment: ", cost_label
    ) |> lapply(htmltools::HTML)
  ) |>
  addLegend(
    pal      = pal,
    values   = ~investment_gap,
    opacity  = 0.7,
    title    = "2027 Missing Stations<br>(Investment Opportunity)",
    position = "bottomright"
  )

print(interactive_map)
