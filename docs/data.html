Veri Kaynakları ve Seçimi

Projemizin temelini oluşturan veri setleri, şeffaflık ve güvenilirlik hedeflenerek iki ana kaynaktan derlenmiştir:

  1. TÜİK (Türkiye İstatistik Kurumu): İl bazında tescilli elektrikli araç sayıları.

Veri Ön İşleme (Preprocessing) ve “Fuzzy Matching”

Veri setlerini birleştirirken karşılaştığımız en büyük engel, kaynaklardaki şehir isimlerinin format uyumsuzluğudur. Bu durumu aşmak ve manuel düzeltme hatalarını sıfıra indirmek için stringdist paketi kullanılarak bir Fuzzy Matching (Bulanık Eşleştirme) algoritması geliştirilmiştir.

# 4. ŞEHİR TAMİR MOTORU (Fuzzy Matching - En Güçlü Yöntem)
clean_and_match_city <- function(messy_city) {
  # Önce tüm çöpü temizle
  cleaned <- messy_city %>%
    str_replace_all("&[[:alpha:]]+;", "") %>% # HTML kodlarını sil
    str_replace_all("[^[:alnum:] ]", " ") %>% # Harf/Rakam dışı her şeyi boşluk yap
    str_replace_all("Ã", "Ç") %>% # Yaygın encoding hataları için ön temizlik
    str_to_lower(locale = "tr") %>%
    str_squish()
  
  # Eğer şehir ismi çok bozuksa (Afyonkara gibi), en yakın resmi ismi bul
  # stringdist paketi burada "en yakın anlamlı kelimeyi" seçer.
  match_idx <- amatch(cleaned, str_to_lower(official_cities, locale = "tr"), maxDist = 5)
  
  if (!is.na(match_idx)) {
    return(official_cities[match_idx])
  } else {
    return(str_to_title(cleaned, locale = "tr"))
  }
}

# 5. Dosya İşleme
process_file <- function(f) {
  tarih_val <- str_extract(basename(f), "\\d{4}_\\d{2}")
  # Encoding bağımsız okuma denemesi
  d <- read.csv(f, stringsAsFactors = FALSE, check.names = FALSE, fileEncoding = "UTF-8")
  if (ncol(d) < 5) d <- read.csv(f, stringsAsFactors = FALSE, check.names = FALSE, fileEncoding = "CP1254")
  
  # Sütun isimlerini zorla ata
  if (ncol(d) == 5) {
    names(d) <- c("sehir", "istasyon", "otomobil", "ev", "oran")
  } else {
    names(d) <- c("sehir", "istasyon", "degisim", "otomobil", "ev", "oran")
  }
  
  d %>%
    as_tibble() %>%
    mutate(
      tarih = tarih_val,
      # Her şehri tek tek resmi liste ile kıyasla ve düzelt
      sehir = map_chr(sehir, clean_and_match_city),
      otomobil_sayisi = as.numeric(str_remove_all(otomobil, "[^0-9]")),
      ev_sayisi = as.numeric(str_remove_all(ev, "[^0-9]")),
      istasyon_sayisi = as.numeric(str_remove_all(istasyon, "[^0-9]")),
      istasyon_doluluk_orani = as.numeric(str_replace_all(oran, ",", "."))
    ) %>%
    select(tarih, sehir, otomobil_sayisi, ev_sayisi, istasyon_sayisi, istasyon_doluluk_orani)
}

# 6. Birleştir ve Kurtar
final_df <- file_list %>%
